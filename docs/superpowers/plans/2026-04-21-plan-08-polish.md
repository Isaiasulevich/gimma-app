- [ ] **Step 2:** Wrap each function's handler.

Create a shared helper `supabase/functions/_shared/sentry.ts`:

```ts
import * as Sentry from '@sentry/deno';

let initialized = false;
export function initSentry() {
  if (initialized) return;
  const dsn = Deno.env.get('SENTRY_DSN_EDGE');
  if (!dsn) return;
  Sentry.init({ dsn, tracesSampleRate: 0.2 });
  initialized = true;
}

export function captureError(e: unknown, extra?: Record<string, unknown>) {
  initSentry();
  Sentry.captureException(e, { extra });
}
```

In each function, `initSentry()` at module load and wrap the `serve` handler:

```ts
import { initSentry, captureError } from '../_shared/sentry.ts';

initSentry();

serve(async (req) => {
  try {
    // ... existing handler body
  } catch (e) {
    captureError(e, { url: req.url });
    throw e;
  }
});
```

```ts
// supabase/functions/_shared/pricing.ts
// USD per 1M tokens (as of 2026-Q1 ā€” revise quarterly).
const PRICING: Record<string, { input: number; output: number }> = {
  'google:gemini-2.0-flash': { input: 0.075, output: 0.30 },
  'google:gemini-2.0-pro':   { input: 1.25,  output: 5.00 },
  'anthropic:claude-opus-4-5':   { input: 3.00,  output: 15.00 },
  'anthropic:claude-sonnet-4-5': { input: 3.00,  output: 15.00 },
  'openai:gpt-4o':           { input: 2.50,  output: 10.00 },
  'openai:gpt-4o-mini':      { input: 0.15,  output: 0.60 },
};

export function estimateCostUsd(
  provider: string,
  model: string,
  inputTokens?: number,
  outputTokens?: number,
): number | null {
  const key = `${provider}:${model}`;
  const p = PRICING[key];
  if (!p) return null;
  const inputCost = ((inputTokens ?? 0) / 1_000_000) * p.input;
  const outputCost = ((outputTokens ?? 0) / 1_000_000) * p.output;
  return +(inputCost + outputCost).toFixed(6);
}
```

## Task 5 ā€” Stall detection

**Files:**
- Modify: `supabase/functions/_shared/compute_metrics.ts`

- [ ] **Step 1:** Flesh out stall detection.

Replace the stub `stalls: []` with real logic:

```ts
// Stall = exercise where best weight hasn't increased in 3+ weeks.
const exerciseBestByWeek: Record<string, Record<string, number>> = {};
// Pull last 5 weeks of sets for this user.
const fiveWeeksAgo = new Date(weekStartDate);
fiveWeeksAgo.setUTCDate(fiveWeeksAgo.getUTCDate() - 28);
const { data: priorSets } = await client
  .from('sets')
  .select('weight, session_exercises!inner(exercise_id, exercises!inner(name), sessions!inner(started_at))')
  .gte('session_exercises.sessions.started_at', fiveWeeksAgo.toISOString())
  .lt('session_exercises.sessions.started_at', weekEnd.toISOString());

for (const s of priorSets ?? []) {
  const row = s as any;
  const exId = row.session_exercises.exercise_id as string;
  const exName = row.session_exercises.exercises.name as string;
  const startedAt = new Date(row.session_exercises.sessions.started_at as string);
  const wk = mondayOfWeek(startedAt);
  exerciseBestByWeek[exId] ??= { __name: exName as any } as any;
  const current = exerciseBestByWeek[exId][wk] ?? 0;
  if ((row.weight ?? 0) > current) exerciseBestByWeek[exId][wk] = row.weight ?? 0;
}

const stalls: WeeklyMetrics['stalls'] = [];
for (const [exId, weeks] of Object.entries(exerciseBestByWeek)) {
  const name = (weeks as any).__name as string;
  const weekEntries = Object.entries(weeks).filter(([k]) => k !== '__name').sort();
  if (weekEntries.length < 3) continue;
  const recent = weekEntries.slice(-3).map(([, v]) => v as number);
  if (recent[0] === recent[1] && recent[1] === recent[2] && recent[0] > 0) {
    stalls.push({ exercise_id: exId, exercise_name: name, weeks_stalled: 3 });
  }
}
```

(Pragmatic implementation ā€” not perfect, but catches obvious 3-week plateaus.)
